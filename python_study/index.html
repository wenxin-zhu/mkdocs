    2.1.随机采样

    def sample(p):
    # np.random.choice 从数组中随机抽取元素
    # 数组p：与数组a相对应，表示取数组a中每个元素的概率，默认为选取每个元素的概率相同
    # 函数功能：按照p数组中的概率，在 0 ~ len(p) - 1 中随机选取一个数
    return np.random.choice(np.arange(p.shape[-1]), p=p)
